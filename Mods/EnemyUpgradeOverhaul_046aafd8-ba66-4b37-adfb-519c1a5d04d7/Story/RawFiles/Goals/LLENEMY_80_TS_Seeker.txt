IF
ObjectEnteredCombat((CHARACTERGUID)_Char, _)
AND
HasActiveStatus(_Char, "LLENEMY_SEEKER", 1)
AND
HasActiveStatus(_Char, "DRUNK", 0)
AND
IsTagged(_Char, "LeaderLib_IsDisabled", 0)
THEN
LeaderLib_Timers_StartObjectTimer(_Char, 1000, "Timers_LLENEMY_Seeker_Pulse", "LLENEMY_Seeker_Pulse");

IF
ObjectLeftCombat((CHARACTERGUID)_Char, _)
AND
HasActiveStatus(_Char, "LLENEMY_SEEKER", 1)
THEN
LeaderLib_Timers_CancelObjectTimer(_Char, "Timers_LLENEMY_Seeker_Pulse");

IF
CharacterStatusApplied(_Char, "DRUNK", _)
AND
HasActiveStatus(_Char, "LLENEMY_SEEKER", 1)
THEN
LeaderLib_Timers_CancelObjectTimer(_Char, "Timers_LLENEMY_Seeker_Pulse");

IF
CharacterStatusRemoved(_Char, "DRUNK", _)
AND
HasActiveStatus(_Char, "LLENEMY_SEEKER", 1)
AND
CharacterIsInCombat(_Char, 1)
AND
IsTagged(_Char, "LeaderLib_IsDisabled", 0)
THEN
LeaderLib_Timers_StartObjectTimer(_Char, 1000, "Timers_LLENEMY_Seeker_Pulse", "LLENEMY_Seeker_Pulse");

IF
ObjectLostTag((CHARACTERGUID)_Char, "LeaderLib_IsDisabled")
AND
HasActiveStatus(_Char, "LLENEMY_SEEKER", 1)
AND
CharacterIsInCombat(_Char, 1)
THEN
LeaderLib_Timers_StartObjectTimer(_Char, 1000, "Timers_LLENEMY_Seeker_Pulse", "LLENEMY_Seeker_Pulse");
